# –°–ø–∏—Å–æ–∫ —Å–æ–æ–±—â–µ–Ω–∏–π
love_messages = [
    "–¢—ã —Å–∞–º–æ–µ –¥–æ—Ä–æ–≥–æ–µ, —á—Ç–æ –µ—Å—Ç—å –≤ –º–æ–µ–π –∂–∏–∑–Ω–∏! ‚ù§Ô∏è",
    "–õ—é–±–ª—é —Ç–µ–±—è –±–æ–ª—å—à–µ, —á–µ–º –∑–≤—ë–∑–¥—ã –Ω–∞ –Ω–µ–±–µ üåü.",
    "–ö–∞–∂–¥—ã–π –¥–µ–Ω—å —Å —Ç–æ–±–æ–π ‚Äî —ç—Ç–æ —Å—á–∞—Å—Ç—å–µ üòä.",
    "–¢—ã –¥–µ–ª–∞–µ—à—å –º–æ–π –º–∏—Ä —è—Ä—á–µ –∏ –ª—É—á—à–µ! üíê",
    "–Ø –Ω–µ —É—Å—Ç–∞—é –≥–æ–≤–æ—Ä–∏—Ç—å —Ç–µ–±–µ, –∫–∞–∫ —Å–∏–ª—å–Ω–æ –ª—é–±–ª—é —Ç–µ–±—è üíï.",
    "–¢—ã –º–æ—ë –≤–¥–æ—Ö–Ω–æ–≤–µ–Ω–∏–µ –∏ –º–æ–π —Å–≤–µ—Ç ‚ú®.",
    "–¢—ã - –ª—É–Ω–∞ –º–æ–µ–π –∂–∏–∑–Ω–∏! üöÄ‚ù§Ô∏è."

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –æ—Ç–ø—Ä–∞–≤–∫–∏ —Å–ª—É—á–∞–π–Ω–æ–≥–æ —Å–æ–æ–±—â–µ–Ω–∏—è
def send_love_message():
    message = random.choice(love_messages)  # –í—ã–±–∏—Ä–∞–µ–º —Å–ª—É—á–∞–π–Ω–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ

    payload = {
        "chat_id": CHAT_ID,
        "text": message
    }
    response = requests.post(url, data=payload)
    if response.status_code == 200:
        print("–°–æ–æ–±—â–µ–Ω–∏–µ —É—Å–ø–µ—à–Ω–æ –æ—Ç–ø—Ä–∞–≤–ª–µ–Ω–æ!")

# –†–∞—Å–ø–∏—Å–∞–Ω–∏–µ –¥–ª—è –æ—Ç–ø—Ä–∞–≤–∫–∏ —Å–æ–æ–±—â–µ–Ω–∏–π
schedule.every().day.at("09:00").do(send_love_message)  # –û—Ç–ø—Ä–∞–≤–∫–∞ –≤ 9:00 —É—Ç—Ä–∞
schedule.every().day.at("21:00").do(send_love_message)  # –û—Ç–ø—Ä–∞–≤–∫–∞ –≤ 21:00 –≤–µ—á–µ—Ä–∞

# –ó–∞–ø—É—Å–∫ –ø–ª–∞–Ω–∏—Ä–æ–≤—â–∏–∫–∞

    while True:
